from geopy import distance
      
    def calc_dist(begin, locs):
      dist = {}
      min_loc = 0
      min = 10000
      sum = 0
      for i,row in locs.iterrows():
        d = distance.distance(begin, (locs.at[i,'lat'],locs.at[i,'long'])).miles
        sum += d
        if d < min:
          min = d
          min_loc = i
          
      dist['min'] = (min_loc,min)
      dist['avg'] = sum/len(locs)
      return dist
